			a = 10*(a)+(b%10);//位置调换
			b=b/10;

			//由于此时a必为正数，因此不考虑符号位，如果在调换的过程中出现某一数据

			if ((b>0) && ((a > 0x7fffffff/10)||(b > 0x7fffffff-a*10)) )
				return 0;
		}
		//最后依据输入数据的大小添加符号位
		return (x>=0)?(int)a:-(int)a;
